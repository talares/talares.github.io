org.talares.api.queries

Query

case class Query(name: String, args: Seq[(String, String)] = Seq())(parent: Option[Query] = None)(options: Seq[QueryOption] = Seq()) extends Product with Serializable

A Scala representation of an OData query URL.

Example:

import org.talares.api.queries._
val q = Query / "Pages" % ("ItemId" -> 123, "PublicationId" -> 1) $ ("Title" ==| "A title") && ("Url" !=| "/path")
